Component Design Best Practices

Build reusable, maintainable components with clear responsibilities

Single Responsibility Principle

❌ Bad: Multiple Responsibilities 

// This component does too many things!
function UserDashboard({ userId }) {
  const [user, setUser] = useState(null);
  const [posts, setPosts] = useState([]);
  const [notifications, setNotifications] = useState([]);
  const [theme, setTheme] = useState('light');
  
  // Fetching user data
  useEffect(() => {
    fetch(`/api/users/${userId}`)
      .then(res => res.json())
      .then(setUser);
  }, [userId]);
  
  // Fetching posts
  useEffect(() => {
    fetch(`/api/posts?userId=${userId}`)
      .then(res => res.json())
      .then(setPosts);
  }, [userId]);
  
  // Theme management
  const toggleTheme = () => {
    setTheme(theme === 'light' ? 'dark' : 'light');
  };
  
  // Rendering everything together
  return (
    <div className={`dashboard ${theme}`}>
      <header>
        <h1>{user?.name}</h1>
        <button onClick={toggleTheme}>Toggle Theme</button>
      </header>
      <main>
        <section>
          {posts.map(post => (
            <div key={post.id}>{post.title}</div>
          ))}
        </section>
        <aside>
          {notifications.map(notif => (
            <div key={notif.id}>{notif.message}</div>
          ))}
        </aside>
      </main>
    </div>
  );
}

✅ Good: Single Responsibility 

// Each component has a single, clear responsibility

// 1. Data fetching hook
function useUser(userId) {
  const [user, setUser] = useState(null);
  
  useEffect(() => {
    fetch(`/api/users/${userId}`)
      .then(res => res.json())
      .then(setUser);
  }, [userId]);
  
  return user;
}

// 2. Theme management
function useTheme() {
  const [theme, setTheme] = useState('light');
  const toggleTheme = () => setTheme(prev => 
    prev === 'light' ? 'dark' : 'light'
  );
  return { theme, toggleTheme };
}

Render Props Pattern 

// Flexible data sharing pattern
function DataProvider({ url, children }) {
  const [data, setData] = useState(null);
  const [loading, setLoading] = useState(true);
  const [error, setError] = useState(null);
  
  useEffect(() => {
    setLoading(true);
    fetch(url)
      .then(res => res.json())
      .then(data => {
        setData(data);
        setError(null);
      })
      .catch(setError)
      .finally(() => setLoading(false));
  }, [url]);
  
  return children({ data, loading, error });
}

// Usage - Consumer decides how to render
function UserList() {
  return (
    <DataProvider url="/api/users">
      {({ data, loading, error }) => {
        if (loading) return <LoadingSpinner />;
        if (error) return <ErrorMessage error={error} />;
        
        return (
          <ul>
            {data.map(user => (
              <li key={user.id}>{user.name}</li>
            ))}
          </ul>
        );
      }}
    </DataProvider>
  );
}

✅ Clean Props API 

// Clear, flexible API with sensible defaults
function Button({
  children,
  variant = 'primary',
  size = 'medium',
  disabled = false,
  loading = false,
  leftIcon,
  rightIcon,
  ...rest
}) {
  const baseClasses = 'btn focus:outline-none transition-colors';
  const variantClasses = {
    primary: 'bg-blue-500 hover:bg-blue-600 text-white',
    secondary: 'bg-gray-200 hover:bg-gray-300 text-gray-800',
    danger: 'bg-red-500 hover:bg-red-600 text-white'
  };
  const sizeClasses = {
    small: 'px-3 py-1 text-sm',
    medium: 'px-4 py-2',
    large: 'px-6 py-3 text-lg'
  };
  
  return (
    <button
      className={`
        ${baseClasses}
        ${variantClasses[variant]}
        ${sizeClasses[size]}
        ${disabled ? 'opacity-50 cursor-not-allowed' : ''}
      `}
      disabled={disabled || loading}
      {...rest}
    >
      {loading && <Spinner size="sm" />}
      {leftIcon && <Icon name={leftIcon} className="mr-2" />}
      {children}
      {rightIcon && <Icon name={rightIcon} className="ml-2" />}
    </button>
  );
}

Naming Conventions & Standards

✅ Good Naming

  • UserProfile - Clear component purpose
  • useUserData - Custom hook prefix
  • handleSubmit - Event handler pattern
  • isLoading - Boolean prefix
  • userId - Clear variable purpose
  • FETCH_SUCCESS - Constant naming

❌ Poor Naming

  • Component1 - Non-descriptive
  • getData - Too generic
  • click - Missing context
  • flag - Unclear boolean
  • temp - Temporary naming
  • data - Too vague

Key Takeaways

  • • Each component should have a single, clear responsibility
  • • Use composition over inheritance for flexible component design
  • • Design props APIs that are intuitive and self-documenting
  • • Follow consistent naming conventions across your codebase
  • • Prefer small, focused components over large, monolithic ones

Modularity & Separation of Concerns

Clear Separation Example 

// features/user-management/services/userAPI.ts
export class UserAPI {
  static async getUsers(params?: GetUsersParams): Promise<User[]> {
    const response = await fetch(`/api/users?${new URLSearchParams(params)}`);
    return response.json();
  }
  
  static async createUser(userData: CreateUserData): Promise<User> {
    const response = await fetch('/api/users', {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify(userData)
    });
    return response.json();
  }
}

// features/user-management/hooks/useUsers.ts
export function useUsers(filters?: UserFilters) {
  const [users, setUsers] = useState<User[]>([]);
  const [loading, setLoading] = useState(true);
  const [error, setError] = useState<string | null>(null);
  
  const fetchUsers = useCallback(async () => {
    try {
      setLoading(true);
      const data = await UserAPI.getUsers(filters);
      setUsers(data);
      setError(null);
    } catch (err) {
      setError(err instanceof Error ? err.message : 'Failed to fetch users');
    } finally {
      setLoading(false);
    }
  }, [filters]);
  
  useEffect(() => {
    fetchUsers();
  }, [fetchUsers]);
  
  return { users, loading, error, refetch: fetchUsers };
}

// features/user-management/components/UserList/UserList.tsx
export function UserList({ filters }: UserListProps) {
  const { users, loading, error } = useUsers(filters);
  
  if (loading) return <LoadingSpinner />;
  if (error) return <ErrorMessage message={error} />;
  
  return (
    <div className="user-list">
      {users.map(user => (
        <UserCard key={user.id} user={user} />
      ))}
    </div>
  );
}

State Management Best Practices

Design efficient state architecture for your applications

Local vs Global State Decision Framework

✅ Use Local State When:

  • State is only used by one component
  • Form input values and validation
  • UI state (modals, dropdowns, tabs)
  • Component-specific loading states
  • Temporary data transformations

📊 Use Global State When:

  • Data needed by multiple components
  • User authentication and profile
  • Application theme and settings
  • Shopping cart contents
  • Real-time notifications

Practical Examples 

// ✅ Good: Local state for component-specific UI
function SearchInput({ onSearch }) {
  const [query, setQuery] = useState('');
  const [isExpanded, setIsExpanded] = useState(false);
  
  const handleSubmit = (e) => {
    e.preventDefault();
    onSearch(query);
  };
  
  return (
    <form onSubmit={handleSubmit}>
      <input
        value={query}
        onChange={(e) => setQuery(e.target.value)}
        onFocus={() => setIsExpanded(true)}
        onBlur={() => setIsExpanded(false)}
        className={isExpanded ? 'expanded' : 'collapsed'}
      />
    </form>
  );
}

// ✅ Good: Global state for shared user data
const UserContext = createContext();

function UserProvider({ children }) {
  const [user, setUser] = useState(null);
  const [isAuthenticated, setIsAuthenticated] = useState(false);
  
  const login = async (credentials) => {
    const userData = await authAPI.login(credentials);
    setUser(userData);
    setIsAuthenticated(true);
  };
  
  const logout = () => {
    setUser(null);
    setIsAuthenticated(false);
    authAPI.logout();
  };
  
  return (
    <UserContext.Provider value={{ user, isAuthenticated, login, logout }}>
      {children}
    </UserContext.Provider>
  );
}

Context API Usage Guidelines

Smart Context Design Pattern 

// Split context by concern to prevent unnecessary re-renders
// ✅ Good: Separate contexts for different concerns

// Theme context - rarely changes
const ThemeContext = createContext();
export function ThemeProvider({ children }) {
  const [theme, setTheme] = useState('light');
  
  const toggleTheme = useCallback(() => {
    setTheme(prev => prev === 'light' ? 'dark' : 'light');
  }, []);
  
  const value = useMemo(() => ({
    theme,
    toggleTheme
  }), [theme, toggleTheme]);
  
  return (
    <ThemeContext.Provider value={value}>
      {children}
    </ThemeContext.Provider>
  );
}

// User context - changes occasionally
const UserContext = createContext();
export function UserProvider({ children }) {
  const [user, setUser] = useState(null);
  const [isLoading, setIsLoading] = useState(true);
  
  const value = useMemo(() => ({
    user,
    isLoading,
    setUser
  }), [user, isLoading]);
  
  return (
    <UserContext.Provider value={value}>
      {children}
    </UserContext.Provider>
  );
}

// Custom hooks for easy consumption
export const useTheme = () => {
  const context = useContext(ThemeContext);
  if (!context) {
    throw new Error('useTheme must be used within ThemeProvider');
  }
  return context;
};

export const useUser = () => {
  const context = useContext(UserContext);
  if (!context) {
    throw new Error('useUser must be used within UserProvider');
  }
  return context;
};

⚠️ Context Anti-patterns to Avoid

  • • Don't put everything in one massive context
  • • Avoid putting frequently changing data in context
  • • Don't use context for component-to-component communication
  • • Avoid creating new objects in context value on every render

State Management Principles

  • • Keep state as close to where it's used as possible
  • • Normalize complex state structures
  • • Use multiple contexts instead of one large context
  • • Implement proper error boundaries for state failures
  • • Consider using state machines for complex state logic
